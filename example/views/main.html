mongoose-asennus

npm install mongoose --save

Schema

var mongoose = require('mongoose');

var personSchema = mongoose.Schema({
    name	        : String,
    job             : String
});

module.exports = mongoose.model('persons', personSchema);

Tiedon käsittely kannasta

var Person = require('./models/person');
